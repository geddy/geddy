 21 /*
 22 AsyncChain -- performs a list of asynchronous calls in a desired order.
 23 Optional "last" method can be set to run after all the items in the
 24 chain have completed.
 25 
 26   // Example usage
 27   var asyncChain = new async.AsyncChain([
 28     {
 29       func: app.trainToBangkok,
 30       args: [geddy, neil, alex],
 31       callback: null, // No callback for this action
 32     },
 33     {
 34       func: fs.readdir,
 35       args: [geddy.config.dirname + '/thailand/express'],
 36       callback: function (err, result) {
 37         if (err) {
 38           // Bail out completely
 39           arguments.callee.chain.abort();
 40         }
 41         else if (result.theBest) {
 42           // Don't run the next item in the chain; go directly
 43           // to the 'last' method.
 44           arguments.callee.chain.shortCircuit();
 45         }
 46         else {
 47           // Otherwise do some other stuff and
 48           // then go to the next link
 49         }
 50       }
 51     },
 52     {
 53       func: child_process.exec,
 54       args: ['ls ./'],
 55       callback: this.hitTheStops
 56     }
 57   ]);
 58 
 59   // Function to exec after all the links in the chain finish
 60   asyncChain.last = function () { // Do some final stuff };
 61 
 62   // Start the async-chain
 63   asyncChain.run();
 64 
 65 */
 66 async.execNonBlocking = function (func) {
 67   if (typeof process != 'undefined' && typeof process.nextTick == 'function') {
 68     process.nextTick(func);
 69   }
 70   else {
 71     setTimeout(func, 0);
 72   }
 73 };

 75 async.AsyncBase = new (function () {
 76 
 77   this.init = function () {
 78     var item;
 79 
 80     this.chain = [];
 81     this.currentItem = null;
 82     this.shortCircuited = false;
 83     this.shortCircuitedArgs = undefined;
 84     this.aborted = false;
 85 
 86     for (var i = 0; i < chain.length; i++) {
 87       item = chain[i];
 88       this.chain.push(new async.AsyncCall(
 89           item.func, item.args, item.callback, item.context));
 90     }
 91   };
 92 
 93   this.runItem = function (item) {
 94     // Reference to the current item in the chain -- used
 95     // to look up the callback to execute with execCallback
 96     this.currentItem = item;
 97     // Scopage
 98     var _this = this;
 99     // Pass the arguments passed to the current async call
100     // to the callback executor, execute it in the correct scope
101     var executor = function () {
102       _this.execCallback.apply(_this, arguments);
103     };
104     // Append the callback executor to the end of the arguments
105     // Node helpfully always has the callback func last
106     var args = item.args.concat(executor);
107     var func = item.func;
108     // Run the async call
109     func.apply(item.context, args);
110   };
111 
112   this.next = function () {
113     if (this.chain.length) {
114       this.runItem(this.chain.shift());
115     }
116     else {
117       this.last();
118     }
119   };
120 
121   this.execCallback = function () {
122     // Look up the callback, if any, specified for this async call
123     var callback = this.currentItem.callback;
124     // If there's a callback, do it
125     if (callback && typeof callback == 'function') {
126       // Allow access to the chain from inside the callback by setting
127       // callback.chain = this, and then using arguments.callee.chain
128       callback.chain = this;
129       callback.apply(this.currentItem.context, arguments);
130     }
131 
132     this.currentItem.finished = true;
133 134 
    // If one of the async callbacks called chain.shortCircuit,
135     // skip to the 'last' function for the chain
136     if (this.shortCircuited) {
137       this.last.apply(null, this.shortCircuitedArgs);
138     }
139     // If one of the async callbacks called chain.abort,
140     // bail completely out
141     else if (this.aborted) {
142       return;
143     }
144     // Otherwise run the next item, if any, in the chain
145     // Let's try not to block if we don't have to
146     else {
147       // Scopage
148       var _this = this;
149       async.execNonBlocking(function () { _this.next.call(_this); });
150     }
151   }
152 
153   // Short-circuit the chain, jump straight to the 'last' function
154   this.shortCircuit = function () {
155     this.shortCircuitedArgs = arguments;
156     this.shortCircuited = true;
157   }
158 
159   // Stop execution of the chain, bail completely out
160   this.abort = function () {
161     this.aborted = true;
162   }
163 
164   // Kick off the chain by grabbing the first item and running it
165   this.run = this.next;
166 
167   // Function to run when the chain is done -- default is a no-op
168   this.last = function () {};
169 
170 })();

172 async.AsyncChain = function (chain) {
173   this.init(chain);
174 };

178 async.AsyncGroup = function (group) {
179   var item;
180   var callback;
181   var args;
182 
183   this.group = [];
184   this.outstandingCount = 0;
185 
186   for (var i = 0; i < group.length; i++) {
187     item = group[i];
188     this.group.push(new async.AsyncCall(
189         item.func, item.args, item.callback, item.context));
190     this.outstandingCount++;
191   }
192 
193 };

195 /*
196 Simpler way to group async calls -- doesn't ensure completion order,
197 but still has a "last" method called when the entire group of calls
198 have completed.
199 */
200 async.AsyncGroup.prototype = new function () {
201   this.run = function () {
202     var _this = this
203       , group = this.group
204       , item
205       , createItem = function (item, args) {
206           return function () {
207             console.log('running');
208             item.func.apply(item.context, args);
209           };
210         }
211       , createCallback = function (item) {
212           return function () {
213             if (item.callback) {
214               item.callback.apply(null, arguments);
215             }
216             console.log('from callback');
217             _this.finish.call(_this);
218           }
219         };
220 
221     for (var i = 0; i < group.length; i++) {
222       item = group[i];
223       callback = createCallback(item);
224       args = item.args.concat(callback);
225       // Run the async call
226       async.execNonBlocking(createItem(item, args));
227     }
228   };
229 
230   this.finish = function () {
231     this.outstandingCount--;
232     console.log('done?');
233     console.log(this.outstandingCount);
234     if (!this.outstandingCount) {
235       this.last();
236     };
237   };
238 
239   this.last = function () {};
240 
241 };

243 var _createSimpleAsyncCall = function (func, context) {
244   return {
245     func: func
246   , args: []
247   , callback: function () {}
248   , context: context
249   };
250 };

252 async.SimpleAsyncChain = function (funcs, context) {
253   chain = [];
254   for (var i = 0, ii = funcs.length; i < ii; i++) {
255 256     chain.push(_createSimpleAsyncCall(funcs[i], context));
257   }
258   this.init(chain);
259 };
260 
261 async.SimpleAsyncChain.prototype = async.AsyncBase;

263 async.AsyncCall = function (func, args, callback, context) {
264   this.func = func;
265   this.args = args;
266   this.callback = callback || null;
267   this.context = context || null;
268 };
