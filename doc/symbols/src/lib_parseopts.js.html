 19 var parseopts = {};
 20 
 21 /**
 22  * @constructor
 23  * Parses a list of command-line args into a key/value object of
 24  * options and an array of positional commands.
 25  * @ param {Array} opts A list of options in the following format:
 26  * [{full: 'foo', abbr: 'f'}, {full: 'bar', abbr: 'b'}]]
 27  28  */
 29 parseopts.Parser = function (opts) {
 30   // Positional commands parse out of the args
 31   this.cmds = [];
 32   // A key/value object of matching options parsed out of the args
 33   this.opts = {};
 34 
 35   // Data structures used for parsing
 36   this.reg = [];
 37   this.shortOpts = {};
 38   this.longOpts = {};
 39 
 40   var item;
 41   for (var i = 0, ii = opts.length; i < ii; i++) {
 42     item = opts[i];
 43     this.shortOpts[item.abbr] = item.full;
 44     this.longOpts[item.full] = item.full;
 45   }
 46   this.reg = opts;
 47 };
 48 
 49 parseopts.Parser.prototype = new function () {
 50 
 51   /**
 52    * Parses an array of arguments into options and positional commands
 53    * Any matcthing opts end up in a key/value object keyed by the 'full'
 54    * name of the option. Any args that aren't passed as options end up in
 55    * an array of positional commands.
 56    * Any options passed without a value end up with a value of null
 57    * in the key/value object of options
 58    * If the user passes options that are not defined in the list passed
 59    * to the constructor, the parser throws an error 'Unknown option.'
 60    * @param {Array} args The command-line args to parse
 61    */
 62   this.parse = function (args) {
 63     var cmds = []
 64       , opts = {}
 65       , arg
 66       , argName
 67       , argItems;
 68 
 69     while (args.length) {
 70       arg = args.shift();
 71       if (arg.indexOf('--') == 0) {
 72         argItems = arg.split('=');
 73         argName = this.longOpts[argItems[0].substr(2)];
 74         if (argName) {
 75           // If there's no attached value, value is true
 76           opts[argName] = argItems[1] || true;
 77         }
 78         else {
 79           throw new Error('Unknown option "' + argItems[0] + '"');
 80         }
 81       }
 82       else if (arg.indexOf('-') == 0) {
 83         argName = this.shortOpts[arg.substr(1)];
 84         if (argName) {
 85           // If there is no following item, or the next item is
 86           // another opt, value is true
 87           opts[argName] = (!args[0] || (args[0].indexOf('-') == 0)) ?
 88               true : args.shift();
 89         }
 90         else {
 91           throw new Error('Unknown option "' + arg + '"');
 92         }
 93       }
 94       else {
 95         cmds.push(arg);
 96       }
 97     }
 98 
 99     this.cmds = cmds;
100     this.opts = opts;
101   };
102 
103 };
104 
105 module.exports = parseopts;
