 19 // This is for documentation purposes only.
 20 // If you are making a router, you should minimally
 21 // provide this interface.
 22 var url = require('url');
 23 
 24 var BaseRouter = function (routes) {
 25   this.routes = [];
 26   if (routes) {
 27     this.setRoutes(routes);
 28   }
 29 };
 30 
 31 /*  This should be in the documented format for your router  */
 32 BaseRouter.prototype = new (function () {
 33   this.setRoutes = function (routes) {
 34     var route
 35       , pattern
 36       , method
 37       , handler;
 38     for (var i = 0, ii = routes.length; i < ii; i++) {
 39       route = routes[i];
 40       // Grab pattern and handler
 41       pattern = route.shift();
 42       handler = route.pop();
 43       // May have a method
 44  45       method = route.shift();
 46       this.match(pattern, method).to(handler);
 47     }
 48   };
 49 
 50   this.match = function (pattern, method) {
 51     var route = new BaseRoute(pattern, method)
 52     this.routes.push(route);
 53     return route;
 54   };
 55 
 56   // Iterate through your routes and return the controller/action
 57   // corresponding to the matched route
 58   this.first = function (req) {
 59     var params = url.parse(req.url, true);
 60     params.controller = null;
 61     params.action = null;
 62     return params;
 63   };
 64 })();
 65 
 66 BaseRoute = function (pattern, method) {
 67   this.pattern = pattern;
 68   this.method = method;
 69   this.handler = null;
 70 };
 71 
 72 BaseRoute.prototype = new (function () {
 73   // Handler should be in format {controller: 'Foo', action: 'bar'}
 74   // If no action is defined, 'handle' action is assumed
 75   this.to = function (handler) {
 76     this.handler = handler;
 77   };
 78 })();
 79 
 80 exports.BaseRouter = BaseRouter;
