 19 /*
 20  * Datatype verification -- may modify the value by casting
 21  */
 22 var datatypes = new (function () {
 23 
 24   var _isArray = function (obj) {
 25     return obj &&
 26       typeof obj === 'object' &&
 27       typeof obj.length === 'number' &&
 28       typeof obj.splice === 'function' &&
 29       !(obj.propertyIsEnumerable('length'));
 30   };
 31 
 32   this.string = function (name, val) {
 33     return {
 34       err: null,
 35  36       val: String(val)
 37     };
 38   };
 39 
 40   this.number = function (name, val) {
 41     if (isNaN(val)) {
 42       return {
 43         err: 'Field "' + name + '" must be a Number.',
 44         val: null
 45       };
 46     };
 47     return {
 48       err: null,
 49       val: Number(val)
 50     };
 51   };
 52 
 53   this.int = function (name, val) {
 54     // Allow decimal values like 10.0 and 3.0
 55     if (Math.round(val) != val) {
 56       return {
 57         err: 'Field "' + name + '" must be an integer.',
 58         val: null
 59       };
 60     };
 61     return {
 62       err: null,
 63       val: parseInt(val, 10)
 64     };
 65   };
 66 
 67   this.boolean = function (name, val) {
 68     var validated;
 69     switch (typeof val) {
 70       case 'string':
 71         if (val == 'true') {
 72           validated = true;
 73         }
 74         else if (val == 'false') {
 75           validated = false;
 76         }
 77         break;
 78       case 'number':
 79         if (val == 1) {
 80           validated = true;
 81         }
 82         else if (val == 0) {
 83           validated = false;
 84         }
 85         break;
 86       case 'boolean':
 87         validated = val;
 88         break;
 89       default:
 90         // Nothing
 91     }
 92 
 93     if (typeof validated != 'boolean') {
 94       return {
 95         err: 'Field "' + name + '" must be a Boolean.',
 96         val: null
 97       };
 98     };
 99     return {
100       err: null,
101       val: validated
102     };
103   };
104 
105   this.object = function (name, val) {
106     // Sure, Arrays are technically Objects, but we're treating Array as a
107     // separate datatype. Remember, instanceof Array fails across window
108     // boundaries, so let's also make sure the Object doesn't have a 'length'
109     // property.
110     if (typeof val != 'object' || _isArray(val)) {
111       return {
112         err: 'Field "' + name + '" must be an Object.',
113 114         val: null
115       };
116     };
117     return {
118       err: null,
119       val: val
120     };
121   };
122 
123   this.array = function (name, val) {
124     // instanceof check can fail across window boundaries. Also check
125     // to make sure there's a length property
126     if (!_isArray(val)) {
127       return {
128         err: 'Field "' + name + '" must be an Array.',
129         val: null
130       };
131     };
132     return {
133       err: null,
134       val: val
135     };
136   };
137 
138   this.date = function (name, val) {
139     var dt = geddy.util.date.parse(val);
140     if (dt) {
141       return {
142         err: null,
143         val: dt
144       };
145     }
146     else {
147       return {
148         err: 'Field "' + name + '" must be in a valid date format.',
149         val: null
150       };
151     }
152   };
153 
154   this.datetime = function (name, val) {
155     var dt = geddy.util.date.parse(val);
156     if (dt) {
157       return {
158         err: null,
159         val: dt
160       };
161     }
162     else {
163       return {
164         err: 'Field "' + name + '" must be in a valid datetime format.',
165         val: null
166       };
167     }
168   };
169 
170   // This is a hack -- we're saving times as Dates of 12/31/1969, and the
171   // desired time
172   this.time = function (name, val) {
173     var dt = geddy.util.date.parse(val);
174     if (dt) {
175       return {
176         err: null,
177         val: dt
178       };
179     }
180     else {
181       return {
182         err: 'Field "' + name + '" must be in a valid time format.',
183         val: null
184       };
185     }
186   };
187 
188 })();
189 
190 module.exports = datatypes;
