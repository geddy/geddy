 19 var fs = require('fs')
 20   , path = require('path');
 21 
 22 var ejs = {};
 23 
 24 ejs.Template = require('./template').Template;
 25 
 26 ejs.textCache = {};
 27 
 28 ejs.TemplateNode = function (id, url, params, parentNode) {
 29   this.id = id;
 30   this.url = url;
 31   this.dirname = path.dirname(url);
 32   this.params = params;
 33   this.parentNode = parentNode || null;
 34   this.loaded = false;
 35   this.cachedFinish = false;
 36   this.childNodes = {};
 37   this.content = null;
 38 };
 39 
 40 ejs.TemplateNode.prototype = new function () {
 41   // Only the root template node gets this -- it's the function that
 42   // passes the completed content back to be written out in the response
 43   this.finishRoot = null;
 44 
 45   this.childFinished = function (childNode) {
 46     this.content = this.content.replace(
 47         '###partial###' + childNode.id, childNode.content);
 48 
 49     // Every time a child node finished, this node may actually
 50     // be finished
 51     this.tryFinish();
 52   };
 53 
 54   this.tryFinish = function () {
 55     // If this node is actually finished, either call up the
 56  57     // chain, or finish the entire thing out
 58     if (this.isFinished()) {
 59       if (this.parentNode) {
 60         this.parentNode.childFinished(this);
 61       }
 62       else {
 63         this.finishRoot();
 64       }
 65     }
 66   };
 67 
 68   var _createLoader = function (node) {
 69     return function () { node.loadTemplate.call(node); };
 70   };
 71 
 72   this.loadChildTemplates = function () {
 73     var childNodes = this.childNodes;
 74     var childNode;
 75     for (var p in childNodes) {
 76       childNode = childNodes[p];
 77       geddy.async.execNonBlocking(_createLoader(childNode));
 78     }
 79   };
 80 
 81   this.isFinished = function () {
 82     // May be a recursive check down into a child -- if the child
 83     // is not loaded, it's not finished
 84     if (!this.loaded) {
 85       return false;
 86     }
 87 
 88     // If we've already closed this node out, use the cached finish-flag
 89     if (this.cachedFinish) {
 90       return true;
 91     }
 92 
 93     // If any of the child nodes of this parent haven't loaded yet,
 94     // bail out -- this will get re-called every time one of them
 95  96     // comes in, until they're all there and we can move on up the chain
 97     var childNodes = this.childNodes;
 98     for (var p in childNodes) {
 99       if (!childNodes[p].isFinished()) {
100         return false;
101       }
102     }
103     // Once this guy is finished cache the results to avoid
104 
105     // a bunch of recursive calls when it's called as a child
106     this.cachedFinish = true;
107 
108     return true;
109   };
110 
111   this.loadTemplate = function () {
112     var _this = this;
113 
114     var handleLoaded = function (data) {
115       _this.loaded = true;
116 
117       // Create a template out of the markup in the file
118       var templ = new ejs.Template({text: data});
119 
120       // Execute the template with the content as the data-object
121       // This may execute "partial" method calls that create child
122       // nodes for this one.
123       templ.process({data: _this.params});
124 
125       _this.content = templ.markup;
126 
127       // By the time we get here, all 'partial' method calls will
128       // have executed, and any children will have been added
129       _this.loadChildTemplates();
130 
131       // This node is loaded, and we now know if it has any children
132       // -- it may be finished, too.
133       _this.tryFinish();
134     };
135 
136     var cached = ejs.textCache[this.url];
137     // Use cached template text if possible
138     if (geddy.config.environment == 'production' && cached) {
139       handleLoaded(cached);
140     }
141     // If this is the first hit, use the template text off disc
142     // and cache it for subsequent requests
143     else {
144       fs.readFile(this.url, 'utf8', function (err, data) {
145         if (err) { throw err; }
146         ejs.textCache[_this.url] = data;
147         handleLoaded(data);
148       });
149     }
150 
151   };
152 
153 }();
154 
155 exports.TemplateNode = ejs.TemplateNode;
