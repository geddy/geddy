 19 var cookies = {};
 20 
 21 cookies.CookieCollection = function (req) {
 22   this.collection = {};
 23   if (req.headers['cookie']) {
 24     this.parse(req.headers['cookie'] || {});
 25   }
 26 };
 27 
 28 cookies.CookieCollection.prototype = new function () {
 29   this.parse = function (header) {
 30     var c = header.split(';');
 31     var item;
 32     var parsed, name, value;
 33     for (var i = 0; i < c.length; i++) {
 34       item = geddy.string.trim(c[i]);
 35       parsed = item.split('=');
 36       name = parsed[0];
 37       value = parsed[1];
 38       this.collection[name] = new cookies.Cookie(name, value);
 39     }
 40   };
 41 
 42   this.toArray = function () {
 43     var send = [];
 44     var c;
 45     for (var p in this.collection) {
 46       c = this.collection[p];
 47       if (c.send) {
 48         send.push(c.toString());
 49       }
 50     }
 51     return send;
 52   };
 53 
 54   this.get = function (name) {
 55     var c = this.collection[name] || {};
 56     return c.value;
 57   };
 58 
 59   this.set = function (name, value, opts) {
 60     var c = new cookies.Cookie(name, value, opts);
 61     c.send = true;
 62     this.collection[name] = c;
 63   };
 64 
 65 }();
 66 
 67 cookies.Cookie = function (name, value, o) {
 68   var opts = o || {};
 69   this.name = name;
 70   this.value = value;
 71   this.path = opts.path || null;
 72   this.expires = opts.expires || null;
 73   this.domain = opts.domain || null;
 74   this.httpOnly = opts.httpOnly || false;
 75   this.send = false;
 76 };
 77 
 78 cookies.Cookie.prototype.toString = function () {
 79   var res = [this.name + '=' + this.value];
 80   var keys = ['path', 'expires', 'domain'];
 81   var key;
 82   var str;
 83   for (var i = 0; i < keys.length; i++) {
 84     key = keys[i];
 85     if (this[key]) {
 86       res.push(key + '=' + this[key]);
 87     }
 88   }
 89   if (this.httpOnly) {
 90     res.push('HttpOnly');
 91   }
 92   str = res.join('; ');
 93   return str;
 94 };
 95 
 96 module.exports = cookies;
