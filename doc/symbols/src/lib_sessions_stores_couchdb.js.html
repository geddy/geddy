 19 var http = require("http");
 20 
 21 var Couchdb = function (callback) {
 22   this.setup(callback);
 23 };
 24 
 25 Couchdb.prototype = new function () {
 26   var _this = this;
 27   var _sessions = {};
 28   var _appCallback;
 29 
 30   this.setup = function (callback) {
 31     _appCallback = callback;
 32     this.request({url: '/' + geddy.config.sessions.dbName,
 33         method: 'GET'}, this.ensureSetup);
 34   };
 35 
 36   this.ensureSetup = function (response) {
 37     if (response.statusCode == 404) {
 38       _this.request({url: '/' + geddy.config.sessions.dbName,
 39           method: 'PUT'}, _this.ensureSetup);
 40     }
 41     else {
 42       _appCallback();
 43     }
 44   };
 45 
 46   this.read = function (session, callback) {
 47     var sid = session.sid;
 48     _appCallback = callback;
 49     this.request({url: '/' + geddy.config.sessions.dbName +
 50         '/' + sid, method: 'GET'}, this.ensureRead);
 51   };
 52 
 53   this.ensureRead = function (response) {
 54     if (response.statusCode == 404) {
 55       _this.create(response);
 56     }
 57     else {
 58       var url = response.url;
 59       var sid = url.substr(url.lastIndexOf('/') + 1);
 60       var data = JSON.parse(response.body);
 61       _sessions[sid] = data;
 62       _appCallback(_sessions[sid]);
 63     }
 64   };
 65 
 66   this.create = function (response) {
 67     var url = response.url;
 68     var sid = url.substr(url.lastIndexOf('/') + 1);
 69     this.request({url: '/' + geddy.config.sessions.dbName +
 70         '/' + sid, method: 'PUT', data: {}}, this.ensureCreate);
 71   };
 72 
 73   this.ensureCreate = function (response) {
 74     if (response.statusCode == 201) {
 75       var url = response.url;
 76       var sid = url.substr(url.lastIndexOf('/') + 1);
 77       var data = JSON.parse(response.body);
 78 
 79       // Avoid another round-trip just to get _id and _rev
 80       delete data.ok;
 81       data._id = data.id;
 82       delete data.id;
 83       data._rev = data.rev;
 84       delete data.rev;
 85 
 86       _sessions[sid] = data;
 87       _appCallback(_sessions[sid]);
 88     }
 89     else {
 90       throw new Error('could not create CouchDB session.');
 91     }
 92   };
 93 
 94   this.write = function (session, callback) {
 95     var sid = session.sid
 96       , data = session.data;
 97     this.request({url: '/' + geddy.config.sessions.dbName +
 98         '/' + sid, method: 'PUT', data: data}, this.ensureUpdate);
 99     _appCallback = callback;
100   };
101 
102   this.ensureUpdate = function (response) {
103     _appCallback();
104   };
105 
106   // FIXME: Convert to use simple clients.couchdb.Client
107   this.request = function (params, callback) {
108     var req = {};
109     req.url = params.url;
110     req.method = params.method || 'GET';
111     req.data = JSON.stringify(params.data) || null;
112 
113     var headers = {host: geddy.config.sessions.hostname};
114     if (req.data) {
115       headers['content-length'] = req.data.length;
116     }
117 
118     var client = http.createClient(geddy.config.sessions.dbPort, geddy.config.sessions.hostname);
119     var request = client.request(req.method, req.url, headers);
120     //console.log(req.url);
121     request.addListener('response', function (response) {
122       //console.log("STATUS: " + response.statusCode);
123       //console.log("HEADERS: " + JSON.stringify(response.headers));
124       response.setEncoding("utf8");
125       response.addListener("data", function (chunk) {
126         //console.log("BODY: " + chunk);
127         callback({
128           url: req.url,
129 130           statusCode: response.statusCode,
131           body: chunk
132         });
133       });
134     });
135     if (req.data) {
136       //console.log('PUT data:');
137       //console.log(req.data);
138       request.write(req.data);
139     }
140     request.end();
141   };
142 
143 
144 }();
145 
146 
147 exports.Couchdb = Couchdb;
